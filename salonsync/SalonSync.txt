// Search
function goToSearch(query) {
  // Pindah ke search screen
  showScreen('screen-search');

  // Letak keyword dalam search input search screen
  var searchInp = document.getElementById('search-input');
  searchInp.value = query;

  // Trigger filter
  filterSearchCards(query);
}

function filterSearchCards(query) {
  query = query.toLowerCase().trim();
  var cards = document.querySelectorAll('#screen-search .salon-card');
  var visibleCount = 0;

  cards.forEach(function(card) {
    var name = card.querySelector('.salon-name').textContent.toLowerCase();
    var loc = card.querySelector('.salon-loc').textContent.toLowerCase();

    var match = query === '' || name.includes(query) || loc.includes(query);
    card.style.display = match ? 'flex' : 'none';
    if (match) visibleCount++;
  });

}

  showScreen('screen-booking');
}

// ── Screen Navigation ─────────────────────────
function showScreen(id) {
  document.querySelectorAll('.screen').forEach(function(s){ s.classList.remove('active'); });
  document.getElementById(id).classList.add('active');
  if (historyStack[historyStack.length-1] !== id) historyStack.push(id);
}

function goBack() {
  if (historyStack.length > 1) {
    historyStack.pop();
    var prev = historyStack[historyStack.length-1];
    document.querySelectorAll('.screen').forEach(function(s){ s.classList.remove('active'); });
    document.getElementById(prev).classList.add('active');
  }
}

function switchNav(el, screenId) {
  document.querySelectorAll('.nav-item').forEach(function(n){ n.classList.remove('active'); });
  el.classList.add('active');
  historyStack = [screenId];
  showScreen(screenId);
}

  document.getElementById('detail-lokasi').style.display = 'none';
  document.getElementById('detail-ulasan').style.display = 'none';
  document.getElementById(tabId).style.display = 'block';
}

// ── Chips / Filters ───────────────────────────
function selectChip(el) {
  el.parentElement.querySelectorAll('.chip').forEach(function(c){ 
    c.classList.remove('active'); 
  });
  el.classList.add('active');

  if (el.closest('#screen-home')) {
    var filter = el.getAttribute('data-filter');
    var cards = document.querySelectorAll('#screen-home .salon-card');

    cards.forEach(function(card) {
      if (filter === 'semua') {
        card.style.display = 'flex';
      } else {
        card.style.display = card.getAttribute('data-location') === filter ? 'flex' : 'none';
      }
    });
  }
}

// Filter
function selFilter(el) {
  el.closest('.filter-row').querySelectorAll('.filter-chip').forEach(function(c){ 
    c.classList.remove('active'); 
  });
  el.classList.add('active');

  var filter = el.getAttribute('data-filter');
  var cards = document.querySelectorAll('#screen-search .salon-card');
  var visibleCount = 0;

  cards.forEach(function(card) {
    var loc = card.getAttribute('data-location');
    var type = card.getAttribute('data-type');
    var show = false;

    if (filter === 'semua') {
      show = true;
    } else if (['kterengganu','kemaman','dungun','marang'].includes(filter)) {
      show = loc === filter;
    } else if (['rambut','spa','kecantikan'].includes(filter)) {
      show = type === filter;
    }

    card.style.display = show ? 'flex' : 'none';
    if (show) visibleCount++;
  });

  var countLabel = document.querySelector('#screen-search .section-title');
  if (countLabel) {
    countLabel.textContent = 'Semua Salon (' + visibleCount + ')';
  }
}

  // Reset filter chips
  document.querySelectorAll('.filter-chip').forEach(function(c){ 
    c.classList.remove('active'); 
  });
  document.querySelector('.filter-chip[data-filter="semua"]').classList.add('active');

  // Update count
  var countLabel = document.querySelector('#screen-search .section-title');
  if (countLabel) {
    countLabel.textContent = query === '' ? 'Semua Salon (4)' : 'Hasil Carian (' + visibleCount + ')';
  }

// ── Toast ─────────────────────────────────────
var _toastTimer;
function showToastMsg(msg) {
  clearTimeout(_toastTimer);
  var el = document.getElementById('toast');
  el.textContent = msg;
  el.classList.add('show');
  _toastTimer = setTimeout(function(){ el.classList.remove('show'); }, 2200);
}

// ── Clock ─────────────────────────────────────
function updateTime() {
  var now = new Date();
  document.querySelector('.status-time').textContent =
    now.getHours().toString().padStart(2,'0') + ':' + now.getMinutes().toString().padStart(2,'0');
}
updateTime();
setInterval(updateTime, 30000);
